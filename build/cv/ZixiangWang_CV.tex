\sectiontitle{Research Interest}
{\small I am a master's student at Peking University , advised by Prof. Liantao Ma. My research focuses on AI for Healthcare , especially medical large language models, electronic health record modeling, multi-agent collaboration, and reusable benchmark/toolkit infrastructure for clinical data analysis.\par}
\begin{enumerate}[label=(\arabic*)]
\item {\small\textbf{Medical Large Language Models}: Building and evaluating LLM-driven systems for clinical prediction, reasoning, retrieval, and decision support.}
\item {\small\textbf{Electronic Health Record Modeling}: Modeling longitudinal, sparse, and multimodal EHR data with attention to clinical reliability and reuse.}
\item {\small\textbf{Benchmarks, Toolkits, and Protocols}: Creating open resources that make healthcare AI evaluation more reproducible and practically useful.}
\end{enumerate}
